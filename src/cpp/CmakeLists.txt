cmake_minimum_required(VERSION 3.25)
project(PyxisProject)

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Include directories
include_directories(${PROJECT_SOURCE_DIR}/include)

# Add source files
set(SOURCES
src/datum_compense.cpp
src/crypto.cpp
)

# Build a static library
add_library(pyxis_static STATIC ${SOURCES})
set_target_properties(pyxis_static PROPERTIES
    OUTPUT_NAME "pyxis"
    POSITION_INDEPENDENT_CODE ON) # Required for shared libraries

# Build a shared library
add_library(pyxis_shared SHARED ${SOURCES})
set_target_properties(pyxis_shared PROPERTIES
    OUTPUT_NAME "pyxis"
    POSITION_INDEPENDENT_CODE ON)

# Create a library
add_library(pyxis ${SOURCES})

# Install targets
install(TARGETS pyxis_static pyxis_shared
        ARCHIVE DESTINATION lib
        LIBRARY DESTINATION lib
        RUNTIME DESTINATION bin)

install(DIRECTORY ${PROJECT_SOURCE_DIR}/include/
        DESTINATION include
        FILES_MATCHING PATTERN "*.h")
